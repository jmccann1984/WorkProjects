        this.AddressNumber = addressNumber;

        this.Street02 = street02;

        this.City = city;
    }

        this.DateCreate = dateCreate;
    }

        return s==null || s.length()==0;
    }

    public String getFullAddress(){
        StringBuilder fullAddress = new StringBuilder();

        if (isNullOrEmpty(getAddressNumber()) && isNullOrEmpty(getStreet()) && isNullOrEmpty(getStreet02()) && isNullOrEmpty(getCity()))
            return "No Address Available";
        else {
            if (!isNullOrEmpty(getAddressNumber()))
                fullAddress.append(getAddressNumber());
            if (!isNullOrEmpty(getStreet())) {
                if (isNullOrEmpty(fullAddress.toString()))
                    fullAddress.append(getStreet());
                else
                    fullAddress.append(" " + getStreet());
            }
            if (!isNullOrEmpty(getStreet02())) {
                if (isNullOrEmpty(fullAddress.toString()))
                    fullAddress.append(getStreet02());
                else
                    fullAddress.append(" " + getStreet02());
            }
            if (!isNullOrEmpty(getCity())) {
                if (isNullOrEmpty(fullAddress.toString()))
                    fullAddress.append(getCity());
                else
                    fullAddress.append(", " + getCity());
            }
            return fullAddress.toString();
        }
    }
}
